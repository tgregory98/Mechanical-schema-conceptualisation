\\
\text{Let $G=(V,E))$ be a graph where $V=V(G),E=E(G)$,}\\
\text{$N,M$ be the sets of 'etl' and 'meta' nodes respectively, where $N\cup M=V(G)$}\\
\text{($N$ stands for \underline{newly} generated nodes, and $M$ stands for \underline{meta} nodes).}\\
\text{Let $n\in N,m\in M$ be singular 'etl' and 'meta' nodes respectively,}\\
$p_m=(\text{\# etl nodes parent to $m$}),c_m=(\text{\# etl nodes child to $m$}),n=\text{(\# etl nodes in total)}.\\
\text{Finally let $\lambda_p,\lambda_c\in\mathbb{R}$ be hyperparameters.}

\\
\text{For each $m\in M$:}\\
\hspace*{1cm}\text{find \% coverage of $m$ given by }\alpha_m:=\frac{\lambda_pp_m+\lambda_cc_m}{n}\\
\hspace*{1cm}\text{then take the reciprocal }\frac{1}{\alpha_m}

\\
\text{For each $n_1,n_2\in N$:}\\
\hspace*{1cm}\text{let }M'_{n_1,n_2}=\{m\in M:n_1m\in E(G)\text{ and }n_2m\in E(G)\}\\
\hspace*{1cm}\text{then }S(n_1,n_2)=\sum_{m\in M'}\frac{1}{\alpha_m}

\\
\text{This yields the structural similarity matrix $S=(S(n_1,n_2))_{n_1,n_2\in N}$ as required.}
